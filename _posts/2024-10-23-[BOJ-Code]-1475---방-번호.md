#include<iostream>
#include<string>
#include<math.h>
using namespace std;

string input_data;
int mem_list[100001] = { 0 };
void init() {
	cin >> input_data;
}
void solve() {

	for (int i = 0; i < input_data.size(); i++) {
		int a = input_data[i] - '0';
		mem_list[a]++;
	}
	int max = 0;
	for (int i = 0; i < 9; i++) {
		int now = mem_list[i];
		if (i == 6) {
			now += mem_list[9];
			now = ceil((double)now / 2);
		}
		if (max < now) {
			max = now;
		}
	}
	cout << max;
}
int main() {
	init();
	solve();

}
