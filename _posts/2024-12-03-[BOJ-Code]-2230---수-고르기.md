#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int N, M;
vector<int> dict;
void init() {
	cin >> N >> M;
	for (int i = 0; i < N; i++) {
		int num;
		cin >> num;
		dict.push_back(num);
	}
	sort(dict.begin(), dict.end());
}
int solve() {
	int en = 0;
	int mn = 2'000'000'001;
	for (int st = 0; st < N; st++) {
		while (en < N && dict[en] - dict[st] < M) en++;
		if (en == N) break;
		mn = min(mn, dict[en] - dict[st]);
	}
	return mn;
}
int main() {
	init();
	cout << solve();
}
