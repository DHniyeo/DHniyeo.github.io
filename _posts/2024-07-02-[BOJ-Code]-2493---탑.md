#include <iostream>
#include <vector>
using namespace std;

int num;
int height[500000];
vector<int> result;

void init() {
    scanf("%d", &num);
    for (int i = 0; i < num; i++) {
        scanf("%d", &height[i]);
    }
}

int main() {
    init();
    vector<int> v; // ì¸ë±ìŠ¤ë¥¼ ì €ì¥í•˜ëŠ” ë²¡í„°

    for (int nowi = 0; nowi < num; nowi++) {
        while (!v.empty() && height[v.back()] <= height[nowi]) {
            v.pop_back();
        }
        if (v.empty()) {
            result.push_back(0);
        } else {
            result.push_back(v.back() + 1);
        }
        v.push_back(nowi);
    }

    for (int i = 0; i < num; i++) {
        printf("%d ", result[i]);
    }

    return 0;
}
