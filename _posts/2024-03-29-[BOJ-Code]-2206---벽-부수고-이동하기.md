#include<stdio.h>
#include<queue>
#include<string.h>
using namespace std;

int n, m;
char map[1000][1001];
int visited[1000][1001][2] = { 0 };

struct info {
	int y, x, dist;
	bool isbroken;
};
int bfs() {
	queue<info> q;

	int dy[] = { -1,0,1,0 };
	int dx[] = { 0,-1,0,1 };
	memset(visited, 0, sizeof(visited));
	// visited [][][0] : ë²½ ì•„ì§ ë¶€ìˆœì ì—†ìŒ
	// visited [][][1] : ë²½ ë¶€ìˆœ ìƒíƒœì—ì„œ ì´ë™í•˜ëŠ” ê²½ìš°
	// visitedë¥¼ í†µí•©í•˜ê²Œ ë˜ë©´ ë²½ì„ ë¶€ìˆ˜ì§€ ì•Šì€ ìƒíƒœì—ì„œ ê°ˆ ìˆ˜ìˆëŠ” ìœ„ì¹˜ë¥¼ ë²½ì„ ë¶€ìˆ´ì„œ ì •ì‘ í•„ìš”í•œ ê²½ìš°ì— ëª» ë¶€ì‹œëŠ” ì¼€ì´ìŠ¤ê°€ íì— ë‹´ê¹€. ë¶€ìˆœ í›„ ì´ë™í•œ ì¼€ì´ìŠ¤ë¥¼ ë¶„ë¦¬í•¨.
	q.push({ 0, 0, 1, 0});
	visited[0][0][0] = 1;

	while (!q.empty()) {
		info now = q.front(); q.pop();
		if (now.y == n - 1 && now.x == m - 1) {
			return now.dist;
		}
		for (int i = 0; i < 4; i++) {
			int ny = now.y + dy[i];
			int nx = now.x + dx[i];
			if (ny >= n || nx >= m || ny < 0 || nx < 0) continue;

			if (map[ny][nx] == '1') { // ë²½ì„ ë§Œë‚œ ê²½ìš°
				if (now.isbroken == true) continue;
				if (visited[ny][nx][1] == 1) continue; // visited 0ì´ë‚˜ 1ì´ë‚˜ ìƒê´€ì—†ìŒ.
				visited[ny][nx][1] = 1;
				q.push({ ny,nx, now.dist + 1, true });
			}
			else if (map[ny][nx] == '0') { // ë²½ì´ ì•„ë‹Œê²½ìš°
				if (now.isbroken == false && visited[ny][nx][0] != 1) { // ë²½ì„ ë¶€ìˆœì  ì—†ê³  ë°©ë¬¸í•œ ì ë„ ì—†ë‹¤ë©´
					visited[ny][nx][0] = 1;
					q.push({ ny,nx, now.dist + 1, false });
				}
				else if(now.isbroken == true && visited[ny][nx][1] != 1) { // ë²½ì„ ë¶€ìˆœìƒíƒœì—ì„œ ë°©ë¬¸í•œ ì ì´ ì—†ë‹¤ë©´
					visited[ny][nx][1] = 1;
					q.push({ ny,nx, now.dist + 1, true});
				}
			}
		}
	}
	return -1;
}
void init() {
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; i++) {
		scanf("%s", map[i]);
	}
}
int main() {
	init();
	int result = bfs();
	printf("%d", result);
}
