#include<stdio.h>


int main() {


	int n;
	int result = 0;
	scanf("%d", &n);
	int cnt = 0;
	for (int i = 666; ; i++) {
		int tmp = i;
		int arr[1000] = {0};
		int k = 0;
		while (tmp / 10) {
			arr[k++] = tmp % 10;
			tmp = tmp / 10;
		}
		arr[k] = tmp;
		int flag = 0;
		for (int x = 0; x <= k-2; x++) {
			if (arr[x] == 6 && arr[x + 1] == 6 && arr[x + 2] == 6) {
				flag = 1;
				break;
			}
		}
		if (flag) cnt++;
		if (cnt == n) {
			result = i;
			break;
		}
	}
	printf("%d", result);
}
