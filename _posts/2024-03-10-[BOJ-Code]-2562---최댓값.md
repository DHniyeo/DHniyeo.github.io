int main()

	}

	int max_v = 0;
	int max_i = 0;
	for (int i = 0; i < 9; i++)
	{
		if (max_v < max[i]) {
			max_v = max[i];
			max_i = i;
		}
	}
	printf("%d\n", max_v);
	printf("%d\n", max_i+1);

}
