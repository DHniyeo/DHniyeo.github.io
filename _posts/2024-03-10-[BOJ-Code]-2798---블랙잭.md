#include <stdio.h>

int n, m;

int main()
{
	scanf("%d %d", &n, &m);
	int arr[100];
	for (int i = 0; i < n; i++) {
		scanf(" %d", &arr[i]);
	}
	int max = 0;
	for (int i = 0; i < n - 2; i++) {
		for (int j = i + 1; j < n - 1; j++) {
			for (int k = j + 1; k < n; k++) {
				if (arr[i] + arr[j] + arr[k] > m) continue;
				if (max < arr[i] + arr[j] + arr[k]) {
					max = arr[i] + arr[j] + arr[k];
				}
			}
		}
	}
	printf("%d", max);
}
