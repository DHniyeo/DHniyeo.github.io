#include<string>

int N;

bool isBad(string c) {
	int size = c.size();

	for (int i = 1; i <= size / 2; i++) {

		if (first == second) return true;
	}
	return false;
}
void dfs(string tmp, int step) {
	if (step == N) {
		cout << tmp << endl;
		exit(0);
		return;
	}
	if(!isBad(tmp+"1")) dfs(tmp + "1", step + 1);
	if(!isBad(tmp+"2")) dfs(tmp + "2", step + 1);
	if(!isBad(tmp+"3")) dfs(tmp + "3", step + 1);
}
int main() {
	scanf("%d", &N);
	dfs("", 0);
}
