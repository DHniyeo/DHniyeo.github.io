#include <iostream>
#include <algorithm>
using namespace std;

int dict_arr[100000];
int find_arr[100000];

int N, M;
void init() {
	cin >> N;
	for (int i = 0; i < N; i++) {
		cin >> dict_arr[i];
	}
	cin >> M;
	for (int i = 0; i < M; i++) {
		cin >> find_arr[i];
	}
	sort(dict_arr, dict_arr + N);
}
int solved(int find_val) {
	int st = 0;
	int en = N - 1;
	while (st<=en) {
		int mid = (st + en) / 2;
		if (dict_arr[mid] < find_val) {
			st = mid+1;
		}
		else if (dict_arr[mid] > find_val) {
			en = mid-1;
		}
		else {
			return 1;
		}
	}
	return 0;
}
int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	init();
	for (int i = 0; i < M; i++) {
		
		int result = solved(find_arr[i]);
		cout << result << "\n";

	}
}

	int st = 0;
	int en = N;
	while (st < en) {
		int mid = (st + en) / 2;
		if (dict[mid] > find_val) {
			en = mid;
		}
		else {
			st = mid + 1;
		}
	}
	return st;
}

	int st = 0;

	while (st < en) {

		if (dict[mid] >= find_val) {
			en = mid;
		}
		else {
			st = mid + 1;
		}
	}
	return st;
}
