int N;
int map[1000][3]; // ì§‘ì„ ìƒ‰ì¹ í•˜ëŠ”ë° ë“œëŠ” ë¹„ìš©
int dp[1000][3];

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < 3; j++) {
			cin >> map[i][j];
		}
	}
}
void solved() {
	for (int i = 0; i < 3; i++) {
		dp[0][i] = map[0][i];
	}
	for (int i = 1; i < N; i++) {
		for (int j = 0; j < 3; j++) {
			int minV = 1e9;
			for (int k = 0; k < 3; k++) {
				if (k == j) continue;
				int tmp = map[i][j] + dp[i - 1][k];
				minV = minV > tmp ? tmp : minV;
			}
			dp[i][j] = minV; // ìµœì†Ÿê°’ì ìš©
		}
	}
	for (int i = 0; i < 3; i++) {
		result = result > dp[N - 1][i] ? dp[N - 1][i] : result;
	}
}
int main() {
	init();
	solved();
	cout << result;
}
