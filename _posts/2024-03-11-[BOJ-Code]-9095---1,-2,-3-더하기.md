#include<stdio.h>
#define min(x,y) x>y?y:x
using namespace std;

int memo[1000001] = {0, 1,2,4};
int dp(int n) {
	if (n == 0 || n == 1 || n==2 || n==3)
		return memo[n];
	for (int i = 4; i <= n; i++) {
		if (memo[i])continue;
		memo[i] = memo[i - 3] + memo[i - 2] + memo[i - 1];
	}
	return memo[n];
}

int main() {
	int tc;
	scanf("%d", &tc);
	getchar();
	for (int T = 0; T < tc; T++) {
		int n;
		scanf("%d", &n);
		getchar();
		
		printf("%d\n", dp(n));
	}



}
