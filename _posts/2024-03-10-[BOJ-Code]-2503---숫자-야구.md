#include<stdio.h>
struct info {
	int num;
	int strike;
	int ball;
};
int main()
{
	int n;
	info arr[100];
	scanf("%d", &n);

	for (int i = 0; i < n; i++)
	{
		scanf(" %d %d %d", &arr[i].num, &arr[i].strike, &arr[i].ball);
	}

	int cnt = 0;
	for (int i = 123; i <= 987; i++) {
		int a = i / 100;
		int b = (i % 100) / 10;
		int c = i % 10;
		if (a == b || b == c || a == c)continue;
		if (a == 0 || b == 0 || c == 0) continue;
		int collect = 0;
		for (int j = 0; j < n; j++) {
			int x = arr[j].num / 100;
			int y = (arr[j].num % 100) / 10;
			int z = arr[j].num % 10;
			int strike = 0;
			int ball = 0;
			if (a == x) strike++;
			if (a == y) ball++;
			if (a == z) ball++;
			if (b == x) ball++;
			if (b == y) strike++;
			if (b == z) ball++;
			if (c == x) ball++;
			if (c == y) ball++;
			if (c == z) strike++;
			if (strike != arr[j].strike || ball != arr[j].ball)break;
			collect++;
		}
		if (collect == n) cnt++;
	}

	printf("%d", cnt);
}
