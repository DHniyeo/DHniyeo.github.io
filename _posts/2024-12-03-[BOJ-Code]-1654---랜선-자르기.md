using namespace std;

int K, N;

			maxV = arr[i];
		}
	}
}
int get_peice(long long length){
	int sum = 0;
	for (int i = 0; i < K; i++) {
		sum += (arr[i] / length);
	}
	return sum;
}

	long long st = 1;
	long long  en = maxV;

		long long  mid = (st + en) / 2;
