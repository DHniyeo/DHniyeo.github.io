```c++
#include<stdio.h>
#include<string.h>


int map[50][50];
int visited[50][50];
int m, n, k;
int cnt;

void input() {
	memset(map, 0, sizeof(map));
	memset(visited, 0, sizeof(visited));
	cnt = 0;
	scanf(" %d %d %d", &m, &n, &k);
	for (int i = 0; i < k; i++) {
		int x, y;
		scanf(" %d %d", &x, &y);
		map[y][x] = 1;
	}
}

const int dy[] = { -1,1,0,0 };
const int dx[] = { 0,0,-1,1 };
void dfs(int y, int x) {
	for (int i = 0; i < 4; i++) {
		int ny = dy[i] + y;
		int nx = dx[i] + x;
		if (ny >= n || nx >= m || ny < 0 || nx < 0) continue;
		if (visited[ny][nx] == 1)continue;
		if (map[ny][nx] != 1) continue;
		visited[ny][nx] = 1;
		dfs(ny, nx);
	}
}

int main() {
	int t;
	scanf(" %d", &t);
	for (int tc = 1; tc <= t; tc++) {
		input();
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < m; j++) {
				if (map[i][j] == 1) {
					if (visited[i][j] == 1) continue;
					dfs(i,j);
					cnt++;
				}
			}
		}
		printf("%d\n", cnt);
	}
}
```
