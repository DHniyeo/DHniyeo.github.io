```c++
#include<stdio.h>
#include<algorithm>

using namespace std;
int map[500][500];
int visited[500][500];
int dp[500][500];
int m, n;
int dy[] = { -1,1,0,0 };
int dx[] = { 0,0,-1,1 };
int dfs_dp(int y, int x) {

	if (y == m-1 && x == n-1 ) return 1;
	if (visited[y][x])return dp[y][x];
	visited[y][x] = 1;
	for (int i = 0; i < 4; i++) {
		int ny = y + dy[i];
		int nx = x + dx[i];
		if (ny < 0 || nx < 0 || ny >= m || nx >= n) continue;
		if (map[y][x] > map[ny][nx]) {
			dp[y][x] += dfs_dp(ny, nx);
		}
	}
	return dp[y][x];
}


int main() {
	
	scanf("%d %d", &m, &n);
	for (int i = 0; i < m; i++) {
		for (int j = 0; j < n; j++) {
			scanf(" %d", &map[i][j]);
		}
	}
	printf("%d",dfs_dp(0,0));


}
```
