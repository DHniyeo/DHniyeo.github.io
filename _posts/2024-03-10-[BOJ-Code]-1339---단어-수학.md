#include <stdio.h>
#include <algorithm>
#include <math.h>
#include <string>

using namespace std;
bool cmp(const int &a, const int &b) {
	return a > b;
}
int main() {
	int n;
	long long memory[100] = {0};
	string arr[10];
	long long result = 0;
	scanf("%d", &n);
	getchar();
	for (int i = 0; i < n; i++) {
		char tmp[11];
		scanf("%s", tmp);
		getchar();
		arr[i] = tmp;
	}

	for (int i = 0; i < n; i++) {
		int string_size = arr[i].size()-1;
		for (int j = 0; j < arr[i].size(); j++) {
			memory[arr[i][j]] +=pow(10, string_size);
			string_size--;
		}
	}
	sort(memory, memory + 100, cmp);

	for (int i = 0; i < 9; i++) {
		result += memory[i] * (9 - i);
	}

	printf("%lld", result);

}
