#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int N;
int dict[1000];
vector<int> two_sum;

void init() {
	cin >> N;
	for (int i = 0; i < N; i++) {
		cin >> dict[i];
	}
	sort(dict, dict + N);
}
int main() {
	init();
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			two_sum.push_back(dict[i] + dict[j]);
		}
	}
	sort(two_sum.begin(), two_sum.end());

	for (int i = N-1; i >=0; i--) {
		for (int j = 0; j <= i; j++) {
			if (binary_search(two_sum.begin(), two_sum.end(), dict[i]-dict[j])) {
				cout << dict[i] << " ";
				return 0;
			}
		}
	}
}
